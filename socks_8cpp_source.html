391 static int replyMethod(int ppipe, unsigned char *inMsg, int method, int *state) {
392  unsigned char outMsg[MAXMSGLENGTH];
393  int msglen = 0;
394 
395  switch (*state) { // respond based on state
396  case NCWAIT:
397  if (method == 0) {
398  memcpy((void *)outMsg, "\x05\x00",2); // build the output message
399  if (wrToPipe(ppipe, (void *)outMsg, 0x02) > 0) { // send the reply
400  *state = SOCKSREQ;
401  return (0); // good send
402  } else {
403  *state = NCWAIT;
404  sockslog->error("replyMethod wrToPipe fail: {}", strerror(errno));
405  return (-1); // if we didn't find good auth method
406  }
407  } else {
408  errno = EPROTO;
409  sockslog->error("replyMethod bad method in NCWAIT");
410  return (-1);
411  }
412  break;
413 
414  case SOCKSREQ:
415  outMsg[SOCKS_VER] = 0x05;
416  outMsg[SOCKS_REP] = (unsigned char) method;
417  outMsg[SOCKS_REP+1] = (unsigned char) 0x00;
418  outMsg[SOCKS_ATYP] = (unsigned char) inMsg[SOCKS_ATYP];
419  if (outMsg[SOCKS_ATYP] == IPV4ADDR) {
420  memcpy((void *)&(outMsg[SOCKS_BNDADDR]), (const void *)&inMsg[SOCKS_BNDADDR],4);
421  outMsg[SOCKS_IPV4BNDPORT]= inMsg[SOCKS_IPV4BNDPORT];
422  outMsg[SOCKS_IPV4BNDPORT+1]= inMsg[SOCKS_IPV4BNDPORT+1];
423  msglen = 10;
424  } else if (outMsg[SOCKS_ATYP] == DOMAINNAMEADDR) {
425  memcpy((void *)&(outMsg[SOCKS_BNDADDR]), (const void *)&
426  inMsg[SOCKS_BNDADDR],inMsg[SOCKS_BNDADDR]+1);
427  outMsg[SOCKS_BNDADDR + 1 + outMsg[SOCKS_BNDADDR]] =
428  inMsg[SOCKS_BNDADDR + 1 + inMsg[SOCKS_BNDADDR]];
429  outMsg[SOCKS_BNDADDR + 1 + outMsg[SOCKS_BNDADDR]+1] =
430  inMsg[SOCKS_BNDADDR + 1 + inMsg[SOCKS_BNDADDR]+1];
431  msglen = 5+outMsg[SOCKS_BNDADDR]+2;
432  } else if (outMsg[SOCKS_ATYP] == IPV6ADDR) {
433  memcpy((void *)&(outMsg[SOCKS_BNDADDR]), (const void *)
434  &inMsg[SOCKS_BNDADDR],16);
435  outMsg[SOCKS_IPV6BNDPORT]= inMsg[SOCKS_IPV6BNDPORT];
436  outMsg[SOCKS_IPV6BNDPORT+1]= inMsg[SOCKS_IPV6BNDPORT+1];
437  msglen = 22;
438  }
439  if (method != 0) {
440  if (wrToPipe(ppipe, (char *)outMsg, msglen) < 0) { // send the reply
441  sockslog->error("replyMethod wrToPipe fail: {}", strerror(errno));
442  }
443  // don't exit here; fall through to the eventLoop and let it exit when
444  // the source resets the connection
445  //{mainlog->info("{} {}:{}: exiting", __PRETTY_FUNCTION__, __FILE__, __LINE__); exit(1);}
446  } else {
447  if (wrToPipe(ppipe, (void *)outMsg, msglen) < 0) { // send the reply
448  sockslog->error("replyMethod wrToPipe fail: {}", strerror(errno));
449  return(-1);
450  }
451  {mainlog->info("{} {}:{}: exiting", __PRETTY_FUNCTION__, __FILE__, __LINE__); exit(1);}
452  }
453  *state = DATAFWRD; // next state
454  break;
455 
456  case DATAFWRD:
457  break;
458 
459  default:
460  *state = NCWAIT;
461  return (NOACCPTMETHOD);
462  break;
