193 void ConnState::updatePeerLoss(seq_no_t ackSeqNo, seq_no_t cumDPRecv) {
194  // FIXME: many ways this can go wrong: wraparound, underflow, init seqNo....
195  seq_no_t losses = ackSeqNo - cumDPRecv;
196  int newLosses = losses - peerLosses;
197  int newDPRecv = cumDPRecv - peerDPRecv;
198 
199  gassert(cumDPRecv > 0);
200  int newLossFloor = newLosses >= 0 ? newLosses : 0;
201  int newDPFloor = newDPRecv >= 0 ? newDPRecv : 0;
202  if (!(newDPFloor | newLossFloor))
203  return;
204  // averaging and playback are equivalent to ~5 decimal places for realistic
205  // ACKs. Likewise, playback permutation is not important in practice,
206  // though can have arbitrarily large effect for large enough loss + new DP.
207  // NB: CTCP playback = 1 DP followed by n losses, which is probably always
208  // wrong.
209 #ifdef PLAYBACK_LOSS
210  if (newLosses > 0)
211  estimatedPeerLoss = estimatedPeerLoss * pow(1 - lossSmoothing, newLosses)
212  + 1 - pow(1 - lossSmoothing, newLosses);
213  if (newDPRecv > 0)
214  estimatedPeerLoss *= pow(1 - lossSmoothing, newDPRecv);
215 #else
216  double newLossAvg = (double) newLossFloor / (newDPFloor + newLossFloor);
217  estimatedPeerLoss = pow(1 - lossSmoothing, newLossFloor + newDPRecv)
218  * (estimatedPeerLoss - newLossAvg) + newLossAvg;
219 #endif
220 
221  peerDPRecv = cumDPRecv;
222  peerLosses = losses;
223 }
