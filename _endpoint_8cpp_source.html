643 // Now tracks writes across calls so as not to busy-loop on EAGAINs. Should
644 // not be called with a different block until the previous one is complete.
645 // TODO: this is almost certainly not how we want to do things: it's simply
646 // expedient.
647 bool Endpoint::sendBlockToForeign(const Block &b) {
648  unsigned char *data_out_raw = (unsigned char *) data_out.get();
649 
650  uint32_t dataLength = 0;
651  int bytes_written = 0;
652 
653  static std::unique_ptr<NonBlockingWriteRequest> wr;
654  static const Block* lastBlock = NULL;
655  if (lastBlock && lastBlock != &b) {
656  endpointlog->critical("Inconsistent foreign send state");
657  endpointlog_p->critical("Endpoint::sendBlockToForeign called with "
658  "conflicting blocks; exiting");
659  {mainlog->info("{} {}:{}: exiting", __PRETTY_FUNCTION__, __FILE__, __LINE__); exit(1);}
660  }
661  if (!lastBlock) {
662  b.decoder->copy_from_symbols(storage::storage(data_out_raw,
663  encoderBlockSize));
664  dataLength = getBlkLength(data_out_raw);
665  endpointlog_p->trace("decoded length = {}", dataLength);
666  if (dataLength > encoderBlockSize - BLK_LENGTH_BYTES) {
667  endpointlog_p->error("Invalid block size {}; discarding block",
668  dataLength);
669  return true;
670  }
671 
672  wr = std::unique_ptr<NonBlockingWriteRequest>(
673  new NonBlockingWriteRequest(socks_socket, data_out, dataLength,
674  BLK_LENGTH_BYTES));
675  lastBlock = &b;
676  }
677  if ((bytes_written = wr->write()) == -1) {
678  endpointlog_p->error("{}: {} {}", __PRETTY_FUNCTION__, strerror(errno), errno);
679  if ((errno == EPIPE) || (errno == EBADF)) {
680  endpointlog->warn("Foreign endpoint has gone away, entering shutdown");
681 // TODO: This a bad way to handle the shutdown in the instance that the source goes way. The shutdown must be
682 // fixed.
683  endpointlog_p->error("{} blastAbort foreign endpoint gone, Exiting", __PRETTY_FUNCTION__);
684  blastAbort();
685  exit(1);
686  cstate.l4s = ConnState::shutdown_pending;
687  socks_socket = -1;
688  }
689  }
690  if (bytes_written > 0) {
691  stats.foreignBytesWritten += bytes_written;
692 #ifdef LOTSA_DEBUG
693  // warning: this is premature now
694  endpointlog->debug("server decoded length = {}", getBlkLength(data_out_raw));
695  endpointlog->trace("server decoded data output to CONTROLWR Bytes = {} : ",
696  bytes_written);
697  pktdumplog->trace(BlockToHexString(b));
698 #endif
699  }
700 
701  if (wr->completed()) {
702  lastBlock = NULL;
703  wr.release();
704  return true;
705  }
706 
707  return false;
708 }
